\newcommand{\newdefinedabbreviation}[3]{
    \newglossaryentry{#1}
    {
      type=\acronymtype, 
      name={#2},
      description={#3},
      first={#3 (#2)},
      plural={{#2}s},
      descriptionplural={{#3}s},
      firstplural={{#3}s ({#2}s)}
    }
}

\newcommand{\newdefinedabbreviationwithplural}[4]{

      description={#3},
      first={#3 (#2)},
      plural={{#2}s},
      descriptionplural={#4},
      firstplural={#4 ({#2}s)}
    }
}
